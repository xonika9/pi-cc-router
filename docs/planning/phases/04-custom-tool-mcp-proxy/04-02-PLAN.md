IMPORTANT: `pi.getAllTools()` is called inside `getCustomToolDefs(pi)`. According to prior findings, this MUST be called lazily — NOT during extension loading. However, the extension's default function IS called during loading. The prior implementation (04-01 SUMMARY) found that `getAllTools()` fails at load time.

To handle this: use a lazy pattern where mcpConfigPath is generated on first request, not at registration time. Declare `mcpConfigPath` as a module-level variable and generate it lazily:

```typescript
let mcpConfigPath: string | undefined;
let mcpConfigResolved = false;

function ensureMcpConfig(pi: ExtensionAPI): string | undefined {
  if (mcpConfigResolved) return mcpConfigPath;
  mcpConfigResolved = true;
  try {
    const toolDefs = getCustomToolDefs(pi);
    if (toolDefs.length > 0) {
      mcpConfigPath = writeMcpConfig(toolDefs);
      console.error(
        `[pi-claude-cli] MCP config generated with ${toolDefs.length} custom tool(s)`,
      );
    }
  } catch (err) {
    console.warn(
      "[pi-claude-cli] MCP config generation failed, custom tools unavailable:",
      err,
    );
  }
  return mcpConfigPath;
}
```

Then in the `streamSimple` callback, call `ensureMcpConfig(pi)` to lazily resolve the config on first request:

```typescript
streamSimple: (model, context, options) => {
  const configPath = ensureMcpConfig(pi);
  return streamViaCli(model, context, { ...options, mcpConfigPath: configPath });
},
```

This pattern:

- Defers getAllTools() until the first LLM request (when pi runtime is fully initialized)
- Generates MCP config once and caches the path
- Handles errors with warn-don't-block (custom tools unavailable, but built-ins still work)

```typescript
import { getModels } from "@mariozechner/pi-ai";
import type { ExtensionAPI } from "@mariozechner/pi-coding-agent";
import { streamViaCli } from "./src/provider.js";
import { validateCliPresence, validateCliAuth } from "./src/process-manager.js";
import { getCustomToolDefs, writeMcpConfig } from "./src/mcp-config.js";

const PROVIDER_ID = "pi-claude-cli";

let mcpConfigPath: string | undefined;
let mcpConfigResolved = false;

function ensureMcpConfig(pi: ExtensionAPI): string | undefined {
  if (mcpConfigResolved) return mcpConfigPath;
  mcpConfigResolved = true;
  try {
    const toolDefs = getCustomToolDefs(pi);
    if (toolDefs.length > 0) {
      mcpConfigPath = writeMcpConfig(toolDefs);
      console.error(
        `[pi-claude-cli] MCP config generated with ${toolDefs.length} custom tool(s)`,
      );
    }
  } catch (err) {
    console.warn(
      "[pi-claude-cli] MCP config generation failed, custom tools unavailable:",
      err,
    );
  }
  return mcpConfigPath;
}

export default function (pi: ExtensionAPI) {
  try {
    validateCliPresence();
    validateCliAuth();

    const models = getModels("anthropic").map((model) => ({
      id: model.id,
      name: model.name,
      reasoning: model.reasoning,
      input: model.input,
      cost: model.cost,
      contextWindow: model.contextWindow,
      maxTokens: model.maxTokens,
    }));

    pi.registerProvider(PROVIDER_ID, {
      baseUrl: "pi-claude-cli",
      apiKey: "unused",
      api: "pi-claude-cli",
      models,
      streamSimple: (model, context, options) => {
        const configPath = ensureMcpConfig(pi);
        return streamViaCli(model, context, {
          ...options,
          mcpConfigPath: configPath,
        });
      },
    });
  } catch (err) {
    console.error(`[pi-claude-cli] Failed to register provider:`, err);
  }
}
```
