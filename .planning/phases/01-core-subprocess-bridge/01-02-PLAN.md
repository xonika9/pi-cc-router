<objective>
Implement the NDJSON stream parser and the event bridge that translates Claude API streaming events into pi's AssistantMessageEventStream format.

Purpose: These two modules form the core of the streaming pipeline. The parser turns raw stdout lines into typed messages, and the event bridge maintains a state machine that tracks content blocks and emits the correct pi events (text_start/delta/end, done) with a properly structured AssistantMessage.

Output: Two independently testable modules with comprehensive unit tests covering happy paths, edge cases, and error resilience.
</objective>

export interface ClaudeStreamEventMessage {
  type: "stream_event";
  event: ClaudeApiEvent;
}

export interface ClaudeResultMessage {
  type: "result";
  subtype: "success" | "error";
  result?: string;
  error?: string;
}

export interface ClaudeSystemMessage {
  type: "system";
  subtype: string;
}

export interface ClaudeControlRequest {
  type: "control_request";
  subtype: "can_use_tool";
  tool_name?: string;
  tool_input?: Record<string, unknown>;
}

export type NdjsonMessage =
  | ClaudeStreamEventMessage
  | ClaudeResultMessage
  | ClaudeSystemMessage
  | ClaudeControlRequest;

export interface ClaudeApiEvent {
  type: string;
  index?: number;
  message?: { usage?: ClaudeUsage; /* ... */ };
  content_block?: { type: string; text?: string; /* ... */ };
  delta?: { type?: string; text?: string; stop_reason?: string; /* ... */ };
  usage?: ClaudeUsage;
}

export interface ClaudeUsage {
  input_tokens?: number;
  output_tokens?: number;
  cache_read_input_tokens?: number;
  cache_creation_input_tokens?: number;
}

export interface TrackedContentBlock {
  type: "text" | "tool_use" | "thinking";
  text: string;
  index: number;
}

- Initialize output with: `role: "assistant"`, `content: []`, `api: "pi-claude-cli"`, `provider: model.provider`, `model: model.id`, zeroed usage, `stopReason: "stop"`, `timestamp: Date.now()`
